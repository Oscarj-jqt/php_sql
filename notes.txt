<?php

class Drink{

    public const SIZE_SMALL = 0;
    public const SIZE_MEDIUM = 1;
    public const SIZE_LARGE = 2;

    private string $name;
    private float $price;
    protected int $size;

    public function __construct(string $name, int $price, int $size){
        $this->name = $name;
        $this->price = $price;
        $this->size = $size;
    }

    public function getPrice() : float {
        return $this->price;
    }
}
